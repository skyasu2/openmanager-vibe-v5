# 32GB 시스템 최적화 WSL 설정 (2025-09-17)
# Claude Code + MCP + 멀티 AI 최고 성능 환경

[wsl2]
# 🌐 네트워킹: MCP 타임아웃 완전 해결
dnsTunneling=true
autoProxy=true  
firewall=true
# networkingMode=mirrored 제거 - 외부 API 안정성 우선

# 💾 메모리: 32GB 시스템의 50% 할당 (최적 비율)
memory=16GB             # 멀티 AI + MCP 대규모 환경 지원
processors=6            # CPU 코어 활용도 극대화
swap=8GB                # 대용량 스왑으로 안정성 확보

# 🚀 고성능 모드 활성화
pageReporting=true      # 메모리 압축으로 효율성 증대
debugConsole=false      # 불필요한 오버헤드 제거
kernelCommandLine=cgroup_no_v1=all systemd.unified_cgroup_hierarchy=1

[experimental]
sparseVhd=true          # 동적 디스크 할당
hostAddressLoopback=true # localhost 성능 최적화
autoMemoryReclaim=gradual # 지능형 메모리 관리
useWindowsDriver=true   # Windows 네이티브 드라이버 사용

# 🎯 16GB 메모리 활용 계획:
#
# 🤖 AI 시스템 (8GB):

# - Gemini CLI: 1.5GB (아키텍처 분석)
# - Qwen CLI: 1.5GB (알고리즘 최적화)
#
# 🔌 MCP 서버들 (4GB):
# - 현재 9개 서버: 3GB
# - 확장 가능 서버: 1GB 예약
#
# 💻 개발 도구들 (3GB):
# - Node.js 빌드: 1.5GB
# - TypeScript 컴파일: 1GB  
# - 테스트 프레임워크: 0.5GB
#
# 🖥️ 시스템 영역 (1GB):
# - Linux 커널 + 서비스
